AZURE Storage:

	AZCOPY:

	b/w Storage account & S3 account:
		Windows		set AWS_ACCESS_KEY_ID=<access-key>
					set AWS_SECRET_ACCESS_KEY=<secret-access-key>
		Linux		export AWS_ACCESS_KEY_ID=<access-key>
					export AWS_SECRET_ACCESS_KEY=<secret-access-key>
		macOS		export AWS_ACCESS_KEY_ID=<access-key>
					export AWS_SECRET_ACCESS_KEY=<secret-access-key>

		Syntax:	azcopy copy 'https://s3.amazonaws.com/<bucket-name>/<object-name>' 'https://<storage-account-name>.blob.core.windows.net/<container-name>/<blob-name>'



	b/w 2 Storage:
		Syntax: azcopy sync '<local-directory-path>' 'https://<storage-account-name>.blob.core.windows.net/<container-name>' --recursive
